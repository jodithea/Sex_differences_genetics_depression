# Start by formatting all sumstats from each cohort consistently
# Do this in an interactive session to check as go
# CHR BP A1 A2 FREQA1 BETA SE P N MAF STUDY INFO
# CHR is a number 1 - 22 and X
# STUDY = 1 in each row - use in meta-analysis to count no. studies meta-analysed


### AGDS ###

AGDS_male=/path/AGDS_4_removed/sex_stratified/GWAS_AGDS_QSKIN_MDD_males_autosomes_and_X_QCed_INFO0.6_MAF0.01.txt

awk 'NR==1 {print "CHR", "BP", "A1", "A2", "FREQA1", "BETA", "SE", "P", "N", "MAF", "STUDY", "INFO"} 
	NR>1 {print $3,$4,$5,$6,$7,$8,$9,$10,$11,$14,"1",$12}' ${AGDS_male} > AGDS_build37_MDD_male_sumstats.txt
# MAF is minor allele freq calculated in all genepi QIMR cohorts

awk '{ gsub("23", "X", $1) ; gsub("25", "X", $1) ; print }' AGDS_build37_MDD_male_sumstats.txt > temp_AGDS.txt && mv temp_AGDS.txt AGDS_build37_MDD_male_sumstats.txt

### Bionic ###

### Blokland et al., 2021 ###

# Use gcOFF = genomic-correction OFF
Blokland21_male_autosomes=/path/Blokland_2021_sumstats/meta_STDERR_mdd_eur_auto_M1_sexstratified_06_gcOFF_pgc+ipsych.txt
Blokland21_male_Xchr=/path/Blokland_2021_sumstats/meta_STDERR_mdd_eur_xchr_M1_sexstratified_06_gcOFF_pgc+ipsych.txt

# Combine autosomes and X chr
cat ${Blokland21_male_autosomes} > Blokland21_build37_MDD_male_sumstats.txt
tail -n +2 ${Blokland21_male_Xchr} >> Blokland21_build37_MDD_male_sumstats.txt

awk 'NR==1 {print "CHR", "BP", "A1", "A2", "FREQA1", "BETA", "SE", "P", "N", "MAF", "STUDY", "INFO"}
        NR>1 {print $16,$17,$2,$3,$4,$8,$9,$10,$18,$22,"1","NA"}' Blokland21_build37_MDD_male_sumstats.txt > temp_Blokland21.txt && mv temp_Blokland21.txt Blokland21_build37_MDD_male_sumstats.txt
# FREQA1 is Overall Frequency Allele 1 across meta-analysis of all cohorts
# MAF is minor allele frequency - not stated in README where from, presumably across all cohorts included in meta-analysis?
# INFO was not provided in sumstats, presumable because imputation score was different in each cohort and they filtered on this before doing the meta-analysis.

awk '{ gsub("23", "X", $1) ; gsub("25", "X", $1) ; print }' Blokland21_build37_MDD_male_sumstats.txt > temp_Blokland21.txt && mv temp_Blokland21.txt Blokland21_build37_MDD_male_sumstats.txt
